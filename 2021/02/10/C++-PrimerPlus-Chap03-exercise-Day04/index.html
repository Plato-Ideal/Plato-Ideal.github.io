  1. 编写一个程序，要求用户使用一个整数指出自己的身高（单位为英寸），然后将身高转换为英尺和英寸。该程序使用下划线表示输入位置。另外使用一个const符号常量表示转换因子；

    /*
编写一个程序，要求用户使用一个整数指出
自己的身高（单位为英寸），然后将身高转换为英尺和英寸。
该程序使用下划线表示输入位置。另外使用一个const符号常量表示转换因子；
*/
#include<iostream>
using namespace std;

int main()
{
    //1英尺=12英寸
    const int feet = 12;//声明一个转换因子 1feet = 12inches
    int height;//user height

    cout << "Please enter your height:____\b\b\b\b";
    cin >> height;//输入为英寸
    cout << endl;
    cout << "Your height is " << height / feet << " feet,";//得到英尺
    cout << "and " << height % feet << " inches." << endl;//得到英寸

    return 0;

}

  1. 编写一个小程序要求以几英尺几英寸的方式输入其身高，并以磅为单位输入其体重，（使用3个变量来存储这些信息），该程序报告其BMI。（BMI = 体重（千克）/ 身高(米) ^ 2）(1英尺 = 12英寸)（1英寸 = 0.0254米）（1千克 = 2.2磅）（影响因子用符号常量表示）

    /*
编写一个小程序要求以几英尺几英寸的方式输入其身高，并以磅为单位输入其体重，（使用3个变量来存储这些信息），该程序报告其BMI。
（BMI = 体重（千克）/ 身高(米) ^ 2）(1英尺 = 12英寸)（1英寸 = 0.0254米）（1千克 = 2.2磅）（影响因子用符号常量表示）
*/
#include<iostream>
using namespace std;

int main()
{
    int feet;//身高 英尺
    int inch;//身高 英寸
    int bounds;//体重 磅

    //声明转换因子
    const int feet_to_inches = 12;
    const double inches_to_meters = 0.0254;
    const double kilometers_to_bounds = 2.2;

    cout << "Enter your height(feet):";
    cin >> feet;
    cout << endl << "Enter your height(inch):";
    cin >> inch;
    cout << endl << "Enter your weight(bounds):";
    cin >> bounds;

    double bmi;//声明BMI
    bmi = (bounds / kilometers_to_bounds) / ((feet * feet_to_inches * inches_to_meters) * (feet * feet_to_inches * inches_to_meters));
    
    cout << "Your BMI is: " << bmi << endl;

    return 0;
}

  1. 编写一个程序，要求用户以度、分、秒的方式输入一个纬度，然后以度为单位显示该纬度。（1度=60分，1分= 60秒，以符号常量表示；对于输入的值应独立存储）

    /*
编写一个程序，
要求用户以度、分、秒的方式输入一个纬度，然后以度为单位显示该纬度。
（1度=60分，1分= 60秒，以符号常量表示；对于输入的值应独立存储）
*/
#include<iostream>
using namespace std;

int main()
{
    //声明符号常量
    const double degree_to_minutes = 60; //度到分钟
    const double minutes_to_seconds = 60; //分钟到秒

    //声明接收变量
    double dergree;
    double minutes;
    double seconds;
    double attitude;

    cout <<"Enter the degree:";
    cin >> dergree;
    cout << endl <<"Enter the minutes:";
    cin >> minutes;
    cout << endl << "Enter the seconds:";
    cin >> seconds;

    double minutes_degrees =  minutes / degree_to_minutes ;
    double seconds_degrees =  seconds / minutes_to_seconds / degree_to_minutes;
    attitude = dergree + minutes_degrees + seconds_degrees;

    cout << "Your attitude is " << attitude << " degrees.";

    return 0;
}

  1. 编写一个程序，要求用户以整数方式输入描述（使用long或long long存储），然后以天、小时、分、秒表示；其换算用符号常量表示；

    /*
编写一个程序，要求用户以整数方式输入秒数（使用long或long long存储），然后以天、小时、分、秒表示；其换算用符号常量表示；
*/
#include<iostream>
using namespace std;

int main()
{
    //声明符号常量
    const int day_hour = 24;
    const int hour_minutes = 60;
    const int minutes_seconds = 60;

    long user_input_seconds;

    cout << "Please enter the seconds:";
    cin >> user_input_seconds;

    int day = user_input_seconds / minutes_seconds / hour_minutes / day_hour;
    int hour = (user_input_seconds - (day * day_hour * hour_minutes * minutes_seconds)) / minutes_seconds / hour_minutes;
    int minutes = (user_input_seconds - ((day * day_hour * hour_minutes * minutes_seconds) + (hour * hour_minutes * minutes_seconds))) / minutes_seconds;
    int seconds = user_input_seconds % minutes_seconds;

    cout << "The " << user_input_seconds << " seconds";
    cout << " equal to " << day << " days and ";
    cout << hour << " hours and";
    cout << minutes << " minutes ";
    cout << "and " << seconds << " seconds.";

    return 0;
}

  1. 编写一个程序，要求用户按欧洲风格输入汽车的耗油量（每百公里耗油（升）），然后将其转换为美国风格（每加仑多少英里）；（100公里 = 62.14英里；1加仑=3.785升）

    /*
编写一个程序，要求用户按欧洲风格输入
汽车的耗油量（每百公里耗油（升）），然后将其转换为美国风格（每加仑多少英里）；
（100公里 = 62.14英里；1加仑=3.785升）
*/
#include<iostream>
using namespace std;

int main()
{
    //定义符号常量
    const double kilometers_miles = 62.14;
    const double gallons_liters = 3.785;

    //接收用户输入
    double gas_consum;
    cout << "Enter your gasoline consumption(Euro Style-How much liters per 1 hundred kilometers):";
    cin >> gas_consum;
    cout << endl;

    double gasllons = gas_consum / gallons_liters;//得到换算成加仑耗油量

    cout << "Your gas consumption(American Style-How much miles per gallon):" << kilometers_miles / gasllons << " miles/gallons";

    return 0;
    /*
    Enter your gasoline consumption(Euro Style-How much liters per 1 hundred kilometers):20
    Your gas consumption(American Style-How much miles per gallon):11.76 miles/gallons
    */

}
