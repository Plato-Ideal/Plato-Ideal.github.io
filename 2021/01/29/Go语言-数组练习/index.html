数组练习

练习一：求数组元素和
//练习1：求数组元素和
var x [7]int
x = [7]int{1, 2, 3, 4, 5, 6, 7}

count := 0
for index := range x {
	count += x[index]
}
fmt.Printf("数组x的元素之和为：%d", count)
练习二：找出数组中和为指定值的两个元素的下标，比如从数组[1,3,5,7]中找出和为8的两个元素的下标分别为（0,3）和（1,2）。
//练习二：找出数组中和为指定值的两个元素的下标，
//比如从数组[1,3,5,7]中找出和为8的两个元素的下标分别为（0,3）和（1,2）。
var x = [8]int{1, 19, 2, 16, 3, 17, 4, 16}
num := 20
flag := true                  //设置标志
for i := 0; i < len(x); i++ { //遍历数组
	for j := len(x) - 1; j > i; j-- {
		if i == j {
			continue //如果i和j相同，就跳过本次循环
		}
		if x[i]+x[j] == num {
			fmt.Printf("(%d,%d)", i, j)
			flag = false
		}
	}

}
if flag { //如果可以进入这个分支结构，说明在for循环中不能找到
	fmt.Printf("找寻不到和为%d的两个元素", num)
}
